  <script>
    const backendInput = document.getElementById('backendUrl');
    const fileInput = document.getElementById('fileInput');
    const analyzeBtn = document.getElementById('analyzeBtn');
    const clearBtn = document.getElementById('clearBtn');
    const thresholdsPre = document.getElementById('thresholds');
    const matchesPre = document.getElementById('matches');

    analyzeBtn.onclick = async () => {
      thresholdsPre.textContent = 'Выполняется...';
      matchesPre.textContent = 'Ожидание...';
      const url = backendInput.value.trim().replace(/\\/$/, '');
      if (!url) return alert('Укажите backend URL.');
      const file = fileInput.files[0];
      if (!file) return alert('Загрузите изображение аудиограммы.');

      const reader = new FileReader();
      reader.onload = async () => {
        try {
          const imageData = reader.result; // dataURL
          // POST /extract
          const resp1 = await fetch(url + '/extract', {
            method: 'POST',
            headers: {'Content-Type':'application/json'},
            body: JSON.stringify({image: imageData})
          });
          if (!resp1.ok) {
            const text = await resp1.text();
            thresholdsPre.textContent = 'Ошибка /extract: ' + resp1.status + ' ' + text;
            matchesPre.textContent = '—';
            return;
          }
          const extracted = await resp1.json();
          thresholdsPre.textContent = JSON.stringify(extracted, null, 2);

          // POST /match
          const resp2 = await fetch(url + '/match', {
            method: 'POST',
            headers: {'Content-Type':'application/json'},
            body: JSON.stringify({thresholds: extracted})
          });
          if (!resp2.ok) {
            const text = await resp2.text();
            matchesPre.textContent = 'Ошибка /match: ' + resp2.status + ' ' + text;
            return;
          }
          const matches = await resp2.json();
          matchesPre.textContent = JSON.stringify(matches, null, 2);
        } catch (e) {
          thresholdsPre.textContent = 'Ошибка: ' + e.message;
          matchesPre.textContent = '—';
        }
      };
      reader.readAsDataURL(file);
    };

    clearBtn.onclick = () => {
      fileInput.value = '';
      thresholdsPre.textContent = '—';
      matchesPre.textContent = '—';
    };
  </script>
